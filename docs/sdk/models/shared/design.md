# Design

Controls how the model applies aesthetics to the generated SVG

## Example Usage

```typescript
import { Design } from "@quiverai/sdk/sdk/models/shared";

let value: Design = {
  colors: {
    palette: [
      "#12FBA2",
      "#1A1A1A",
      "#FFFFFF",
    ],
    roles: {
      accent: [
        "#12FBA2",
      ],
      background: [
        "#FFFFFF",
      ],
      primary: [
        "#1A1A1A",
      ],
    },
  },
  stylePreset: "outline",
};
```
