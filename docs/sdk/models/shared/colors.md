# Colors

The color system configuration for SVG generation

## Example Usage

```typescript
import { Colors } from "@quiverai/sdk/sdk/models/shared";

let value: Colors = {
  palette: [
    "#12FBA2",
    "#1A1A1A",
    "#FFFFFF",
  ],
  roles: {
    accent: [
      "#12FBA2",
    ],
    background: [
      "#FFFFFF",
    ],
    primary: [
      "#1A1A1A",
    ],
  },
};
```
